SQL CREATE VIEW Statement

SQL - Create a Simple View

This query creates a view named employee_salaries that shows employee names and their salaries from the employees table. 

CREATE VIEW employee_salaries AS
SELECT employee_name, salary
FROM employees;

SQL - Create a View with Joins

This query creates a view named department_salary_view that shows department names along with the total salaries for each department. 

CREATE VIEW department_salary_view AS
SELECT d.department_name, SUM(e.salary) AS total_salary
FROM departments d
JOIN employees e
ON d.department_id = e.department_id
GROUP BY d.department_name;

SQL - Create a View with Conditions

This query creates a view named high_salary_employees that shows employees with salaries greater than $80,000. 

CREATE VIEW high_salary_employees AS
SELECT employee_name, salary
FROM employees
WHERE salary > 80000;
