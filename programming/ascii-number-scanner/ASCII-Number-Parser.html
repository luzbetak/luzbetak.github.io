ASCII Policy Number Parser

This Python script reads policy numbers in ASCII format from a file, parses them, validates their checksum, and writes the results to a new file.

Code Explanation


#!/usr/bin/env python

from checksum import calculate_checksum

# ------------------------------------------------------------------------------------------------ #
# Define ASCII-to-digit mappings for numbers 0-9
# This dictionary maps ASCII representations of digits (as tuples of 3 lines) to their actual
# digit values.
# ------------------------------------------------------------------------------------------------ #
ascii_to_digit = {
    (" _ ", "| |", "|_|"): "0",
    ("   ", "  |", "  |"): "1",
    (" _ ", " _|", "|_ "): "2",
    (" _ ", " _|", " _|"): "3",
    ("   ", "|_|", "  |"): "4",
    (" _ ", "|_ ", " _|"): "5",
    (" _ ", "|_ ", "|_|"): "6",
    (" _ ", "  |", "  |"): "7",
    (" _ ", "|_|", "|_|"): "8",
    (" _ ", "|_|", " _|"): "9",
}

# ------------------------------------------------------------------------------------------------ #
def parse_ascii_policy_numbers(file_path):
    """
    Reads the ASCII representation of all policy numbers from the given file.
    
    Args:
        file_path (str): The path to the file containing ASCII-formatted policy numbers.
        
    Returns:
        list of tuples: A list containing tuples, where each tuple consists of the parsed policy 
        number as a string and a boolean indicating if it contains any unrecognized characters.
    """
    with open(file_path, "r") as f:
        lines = [line.rstrip("\n") for line in f]  # Read and strip newline characters from each line

    policy_numbers = []

    # Process the file in blocks of 4 lines (3 lines of digits + 1 blank line)
    for i in range(0, len(lines), 4):
        entry = lines[i:i + 3]  # Extract the 3 lines representing a policy number

        # Check if the entry has the expected 3 lines
        if len(entry) < 3:
            print(f"Skipping incomplete entry at lines {i}-{i+3}.")
            continue

        digits = []
        contains_unrecognized = False  # Track if there's a "?" digit

        # Extract each 3-character block for all 9 digits
        for j in range(0, 27, 3):
            digit_tuple = (entry[0][j:j+3], entry[1][j:j+3], entry[2][j:j+3])
            digit = ascii_to_digit.get(digit_tuple, "?")

            # --- Unrecognized digit === #
            if digit == "?":
                contains_unrecognized = True  # Mark as containing invalid digits
            digits.append(digit)

        policy_numbers.append(("".join(digits), contains_unrecognized))  # Combine digits into a policy number

    return policy_numbers

# ------------------------------------------------------------------------------------------------ #
if __name__ == "__main__":
    # Main execution starts here.
    file_path = "policy_numbers_bronze.txt"  # The input file with ASCII policy numbers
    parsed_policy_numbers = parse_ascii_policy_numbers(file_path)  # Parse the policy numbers

    with open("policy_numbers_silver.txt", "w") as output_file:
        # Process each parsed policy number
        for policy_number, contains_unrecognized in parsed_policy_numbers:
            if contains_unrecognized:
                output = f"{policy_number} ILL"  # Mark policy as illegal (contains unrecognized digits)
            elif calculate_checksum(policy_number):
                output = policy_number  # Write valid policy number
            else:
                output = f"{policy_number} ERR"  # Mark policy as erroneous (checksum failed)

            print(output)  # Print the result to the console
            output_file.write(output + "\n")  # Write the result to the output file

How to Run the Script

This script reads policy numbers from a file policy_numbers_bronze.txt, parses the ASCII digits, validates the checksum, and writes the results to a new file policy_numbers_silver.txt.

Steps to Run:

  1. Make sure you have a valid checksum.py file with the calculate_checksum function.
  2. Create or provide the policy_numbers_bronze.txt file containing the ASCII policy numbers.
  3. Run the script using the command:

    python3 parse_policy_numbers.py

Expected Output:

  • If the policy number contains invalid characters, it is marked with ILL.
  • If the checksum fails, it is marked with ERR.
  • Valid policy numbers are printed and saved as-is.
